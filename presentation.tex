\section{Introduction}
% ==============================================================================

\begin{frame}{Motivation}
  \begin{itemize}
    \item Organizations strive for fast flow through autonomous teams
    \item Architecture decisions often remain centralized in a few experts
    \item Architectural uncertainty is natural but usually \alert{implicit}
    \item[]
    \item Existing derivation methods rely on architects' tacit knowledge
          (Souza et al., 2019)
    \item Teams without senior architects lack guidance and tooling
  \end{itemize}
\end{frame}

\begin{frame}{The Problem}
  \begin{block}{Gap}
    No consolidated approach exists for managing architectural uncertainty
    within the daily engineering workflow of development teams.
  \end{block}

  \vspace{0.5\baselineskip}

  \begin{itemize}
    \item ArchHypo (Silva et al., TSE 2025) uses hypotheses engineering
          to make uncertainty explicit and manageable
    \item But: manual application has a \alert{steep learning curve}
          (Finding 9: unanimously hard to learn)
    \item Process-change resistance (Finding 8) hinders adoption
  \end{itemize}
\end{frame}

\begin{frame}{Objective}
  \begin{block}{Thesis Objective}
    Empirically evaluate and refine \textbf{Hypo-Stage}, a support tool for
    the ArchHypo technique, by applying it with diverse software engineering
    teams in a real-world organization with a scaled software product.
  \end{block}

\section{Background}
% ==============================================================================

\begin{frame}{ArchHypo Technique}
  \begin{enumerate}
    \item \textbf{Identify} architectural hypotheses (uncertainties)
    \item \textbf{Assess} each: uncertainty level + impact level
    \item \textbf{Plan} technical actions (spikes, tracer bullets, triggers)
    \item \textbf{Execute} plans and \textbf{reassess} iteratively
  \end{enumerate}

  \vspace{0.5\baselineskip}

  \begin{itemize}
    \item Prior evaluations: Catch Solve startup (IEEE SW), PropSEP at Jetsoft (TSE)
    \item Both manual --- no tool support
  \end{itemize}
\end{frame}

\begin{frame}{Hypo-Stage Tool}
  \begin{itemize}
    \item Open-source Backstage plugin for ArchHypo
    \item Hypothesis CRUD, Likert-scale assessment, technical plans
    \item Visualization: uncertainty vs. impact chart
    \item Integrates into the Internal Developer Portal workflow
  \end{itemize}

  \vspace{0.5\baselineskip}

  \begin{block}{Lineage}
    Initial design \& implementation: undergrad capstone (Corrêa, 2025).\\
    This thesis: empirical evaluation \& refinement with real teams.
  \end{block}

\appendix

\begin{frame}{Backup: ArchHypo Nine Findings (TSE)}
  \footnotesize
  \begin{enumerate}
    \item Some hypotheses monitored without full resolution until end
    \item Avoided upfront architectural design; decisions through iterations
    \item Improved process: safety, transparency, predictability
    \item Different QAs handled with different creative strategies
    \item Guideline implementation was the most-used strategy
    \item Team recognized better architectural decision-making
    \item Evidence of positive impact on product quality
    \item \alert{Process-change obstacle}: mixing arch \& functional tasks
    \item \alert{Unanimously hard to learn}: mapping, articulating, planning
  \end{enumerate}
\end{frame}

\begin{frame}{Backup: Souza et al. Key Findings}
  \begin{itemize}
    \item 2575 papers $\rightarrow$ 39 primary studies
    \item All methods rely on architects' tacit knowledge
    \item Limited support for inexperienced practitioners
    \item Weak or absent tool support
    \item Insufficient empirical evaluation
  \end{itemize}
\end{frame}
